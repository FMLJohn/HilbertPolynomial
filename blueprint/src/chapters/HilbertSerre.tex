\chapter{The Hilbert--Serre Theorem}\label{HilbertSerre}

The statement of the Hilbert--Serre Theorem relies on these assumptions: $A=\oplus_{n=0}^{\infty}{A}_n$ is a Noetherian graded ring, $a_1,\ldots,a_s$ are nonzero homogeneous elements of $A$ of degrees $d_1,\ldots,d_s>0$ such that $A=A_0[a_1,\ldots,a_s]$, and $M=\oplus_{n=0}^{\infty}M_n$ is a finitely generated graded module over $A$. In order to state the Hilbert--Serre Theorem, we need to first prove some preliminary lemmas and define some additional concepts.

\begin{lemma}
\label{gradezeroNoering}
\lean{GradedRing.GradeZero.isNoetherianRing}
\leanok
$A_0$ is a subring of $A$ which is itself Noetherian.
\end{lemma}
\begin{proof}
By definition of a graded ring, the degree-zero component is closed under addition and multiplication, and it contains $0$ and $1$, so $A_0$ is a ring. Also, $A_0$ is isomorphic to the quotient $A/\oplus_{n=1}^{\infty}A_n$. As a quotient ring of a Noetherian ring is again Noetherian, $A_0$ is Noetherian.
\end{proof}

\begin{lemma}
\label{gradeifg}
\uses{gradezeroNoering}
lean{GradedModule.finite_module_over_degree_zero_subring}
\leanok
For any $n\in\{0,1,\ldots\}$, $M_n$ is a finitely generated module over $A_0$.
\end{lemma}
\begin{proof}
Pick some $n\in\{0,1,\ldots\}$. $M_n$ is a module over $A_0$, because $M_n$ is closed under addition, and multiplication by an element of $A_0$ does not change the degree of any element in $M_n$. We still need to show that $M_n$ is finitely generated over $A_0$. Let $m_1,\ldots,m_l\in{M}$ be some homogeneous elements of degrees $c_1,\ldots,c_l$ that generate $M$ as an $A$-module. Let $C_n$ be the collection of all elements in $M_n$ that are of the form $p{\cdot}m_j$ ($j=1,\ldots,l$), where the degree of $m_j$ is not greater than $n$, and $p=a_1^{n_1}\cdots{a_s}^{n_s}$ is a monomial of degree $n-c_j$. As we have assumed that $a_1,\ldots,a_s$ have positive degrees, $C_n$ is a finite set. Moreover, each element of $M_n$ also belongs to $M$, and $M$ is generated by the $m_j$'s, so $C_n$ generates $M_n$ as an $A_0$-module.
\end{proof}

\begin{definition}
\label{exactseq}
Let $R$ be a ring. A sequence of $R$-modules and $R$-module homomorphisms
\begin{equation}
\cdots\stackrel{f_{-1}}{\rightarrow}M_0\stackrel{f_0}{\rightarrow}M_1\stackrel{f_1}{\rightarrow}M_2\stackrel{f_2}{\rightarrow}\cdots
\end{equation}
is \textbf{exact} if for any $f_i$ and $f_{i+1}$ in the sequence, the image of $f_{i}$ equals the kernel of $f_{i+1}$.
\end{definition}

\begin{definition}
\label{shortexactseq}
Let $R$ be a ring. A \textbf{short exact sequence} of $R$-modules and $R$-module homomorphisms is an exact sequence
\begin{equation}
\label{shortexaseq}
0{\rightarrow}M_1{\rightarrow}M_2{\rightarrow}M_3{\rightarrow}0
\end{equation}
where $0$ is the trivial module over $R$.
\end{definition}

\begin{definition}
\label{moduleaddfunc}
\uses{shortexactseq}
Let $R$ be a ring, $C$ be a class of modules over $R$ and $(G,+)$ be an abelian group. An \textbf{additive function} on $C$ with values in $G$ is a map $\mu:C\rightarrow{G}$ such that whenever there is a short exact sequence (\ref{shortexaseq}) with $M_1,M_2,M_3\in{C}$, we have that $\mu(M_1)+\mu(M_3)=\mu(M_2)$.
\end{definition}

Additive functions can be viewed as a generalisation of the dimensions of finite-dimensional vector spaces. Indeed, let $F$ be a field. Given any short exact sequence $0\rightarrow{V_1}\rightarrow{V_2}\rightarrow{V_3}\rightarrow{0}$ of finite-dimensional vector spaces over $F$, we have that $V_3$ is isomorphic to $V_2/V_1$, so $\operatorname{dim}(V_1)+\operatorname{dim}(V_3)=\operatorname{dim}(V_2)$.

It is well-known that for any commutative Noetherian ring $R$, the class of finitely generated $R$-modules forms an abelian category. Hence, we decided to formalise the definition of additive functions as the following:

\begin{definition}
\label{abeliancataddfunc}
\lean{AdditiveFunction}
\leanok
Given an abelian category $\mathcal{C}$ and an additive commutative group $G$, a function $\mu$ from the class of all objects of $\mathcal{C}$ to $G$ is \textbf{additive} if for any short exact sequence $0\rightarrow{A}\rightarrow{B}\rightarrow{C}\rightarrow{0}$ with elements in $\mathcal{C}$, we have $\mu(A)+\mu(C)=\mu(B)$.
\end{definition}

\begin{definition}
\label{Poincareser}
\lean{AdditiveFunction.poincareSeries}
\leanok
\uses{gradezeroNoering, gradeifg, moduleaddfunc, abeliancataddfunc}
Let $C$ be the class of all finitely generated modules over $A_0$ and suppose $\mu$ is an additive function on $C$ with values in $\mathbb{Z}$. Then the \textbf{Poincar\'e series} of $M$ in terms of $\mu$ is the formal power series
\begin{equation}
P(M,\mu,t)=\sum_{n=0}^{\infty}\mu(M_n)t^n\hspace{1mm}.
\end{equation}
\end{definition}

Now we are ready to state the Hilbert--Serre Theorem:

\begin{theorem}[Hilbert--Serre]
\label{HilSer}
\uses{gradezeroNoering, Poincareser, additivezero, sumeq0ofexact, KLprop, KLfg}
\lean{hilbert_serre}
\leanok
Let $\mu$ be an additive function on the class of all finitely generated $A_0$-modules with values in $\mathbb{Z}$. Then $P(M,\mu,t)$ can be written as
\begin{equation}
\frac{p(t)}{\Pi_{i=1}^s(1-t^{d_i})}
\end{equation}
for some polynomial $p$ with integer coefficients.
\end{theorem}

The Hilbert--Serre Theorem means that some combinatorial phenomenon has occurred among the coefficients of $P(M,\mu,t)$ and $\Pi_{i=1}^s(1-t^{d_i})$ --- after an expression with infinitely many terms is multiplied by a polynomial, only a finite number of terms remain. We can equivalently state the theorem as ``the product of $P(M,\mu,t)$ and $\Pi_{i=1}^s(1-t^{d_i})$ is a polynomial in $\mathbb{Z}[t]$''. A simpler example of this phenomenon is the power series expansion of $1/(1-t)$, which is $1+t+t^2+\cdots$. It is easy to see that after we multiply $1+t+t^2+\cdots$ by $1-t$, every term other than $1$ vanishes. This fact will be quite useful for defining Hilbert polynomials. For now we focus on how to prove the Hilbert--Serre Theorem. We still need some lemmas about additive functions.

\begin{lemma}
\label{additivezero}
\uses{moduleaddfunc, abeliancataddfunc}
\lean{AdditiveFunction.map_zero}
\leanok
Let $R$ be a ring and let $\mu$ be an additive function on some class of $R$-modules that contains the trivial module $0$. Then $\mu(0)=0$, where the second zero is the identity element of the relative group.
\end{lemma}
\begin{proof}
We have a short exact sequence of $R$-modules $0\rightarrow{0}\rightarrow{0}\rightarrow{0}\rightarrow{0}$, implying that $\mu(0)+\mu(0)=\mu(0)$. Hence $\mu(0)$ must be zero.
\end{proof}

\begin{lemma}
\label{sumeq0ofexact}
\uses{exactseq, moduleaddfunc, abeliancataddfunc, additivezero}
\lean{AdditiveFunction.alternating_sum_apply_eq_zero_of_zero_zero_of_length6'}
\leanok
Let $R$ be a ring and suppose $\mu$ is an additive function on some class $C$ of $R$-modules. Suppose we have an exact sequence
\begin{equation}
0\stackrel{f_1}{\rightarrow}M_1\stackrel{f_2}{\rightarrow}M_2\stackrel{f_3}{\rightarrow}{M_3}\stackrel{f_4}{\rightarrow}{M_4}\stackrel{f_5}{\rightarrow}0,
\end{equation}
where $M_1,\ldots,M_4\in{C}$ and for each $i=1,\ldots,5$, the kernel of $f_i$ belongs to $C$. Then
\begin{equation}
\mu(M_1)-\mu(M_2)+\mu(M_3)-\mu(M_4)=0.
\end{equation}
\end{lemma}
\begin{proof}
For each $i\in\{1,\ldots,5\}$, let $I_i$ be the image of the map $f_i$. Then for any $i\in\{1,\ldots,5\}$, there is a short exact sequence $0\rightarrow{I_i}\rightarrow{M_i}\rightarrow{I_{i+1}}\rightarrow{0}$, so by the definition of an additive function we have that $\mu(M_i)=\mu(I_i)+\mu(I_{i+1})$. Note that $\mu(I_1)=\mu(I_{5})=0$ because both $I_1$ and $I_{5}$ are the trivial module. Hence
\begin{equation}
\begin{split}
& \mu(M_1)-\mu(M_2)+\mu(M_3)-\mu(M_4)
\\
& = [\mu(I_1)+\mu(I_2)]+[-(\mu(I_2)+\mu(I_3))]+[\mu(I_3)+\mu(I_4)]+[-(\mu(I_4)+\mu(I_{5}))]
\\
& = \mu(I_1)-\mu(I_{5})
\\
& = 0.
\end{split}
\end{equation}
\end{proof}

Note that Lemma \ref{sumeq0ofexact} can be generalised to exact sequences of any length that start and end with the trivial module, but since the longest sequence in our application contains no more than four nontrivial terms, we are content with the current version. Now we are ready to prove the Hilbert--Serre Theorem. We prove it by induction on $s$; we now restate the theorem in the precise form in which we prove it:

\begin{theorem}[Hilbert--Serre, restatement]
\label{HilSerInd}
Suppose $R$ is a ring and $\mu$ is an additive function on the class of all finitely generated $R$-modules with values in $\mathbb{Z}$. Then for any $s\in\mathbb{N}$, if $A=\oplus_{n=0}^{\infty}A_n$ is a Noetherian graded ring with $A_0=R$, $M=\oplus_{n=0}^{\infty}M_n$ is a finitely generated $A$-module, and $A=A[a_1,\ldots,a_s]$ for some nonzero homogeneous elements $a_1,\ldots,a_s\in{A}$, then
\begin{equation}
P(M,\mu,t)=\frac{p(t)}{\Pi_{i=1}^s(1-t^{d_i})}
\end{equation}
for some polynomial $p$ with coefficients in $\mathbb{Z}$.
\end{theorem}

\begin{proof}[Proof ($s=0$ case)]
If $s=0$, then $A=A_0$. As $M$ is a finitely generated $A_0$-module, there exist homogeneous elements $m_1,\ldots,m_l\in{M}$ that generate $M$. Let $k$ be the highest degree of an element in $\{m_1,\ldots,m_l\}$. Then the degrees of the homogeneous components of any element in $M$ cannot exceed $k$, which implies that $M_n$ is the zero module for any $n>k$. By Lemma \ref{additivezero}, if we write out the Poincar\'e series $P(M,\lambda,t)$, then there will only be finitely many terms, meaning that the series is a polynomial in $t$, so we are done.
\end{proof}

Now we assume Theorem \ref{HilSerInd} holds for $s$ and try to prove that it also holds for $s+1$. Note that for any $n\in{\mathbb{N}}$, the map $g_n$ defined by $m\mapsto{a_{s+1}}m$ is an $A_0$-module homomorphism from $M_n$ to $M_{n+d_{s+1}}$, where $d_{s+1}$ is the degree of $a_{s+1}$. We define $K_n$ and $L_{n+{d_{s+1}}}$ as $\operatorname{ker}(g_n)$ and $M_{n+d_{s+1}}/im(g_n)$, respectively. Let $K=\oplus_{n=0}^{\infty}K_n$ and let $L$ be the quotient module $M/a_{s+1}M$. Then $L$ is a graded $A$-module such that for any natural number $n\ge{d_{s+1}}$, the $n$-th grade is $L_{n}$.

\begin{lemma}
\label{KLprop}
$K$ and $L$ are finitely generated graded modules over $A$ such that $a_{s+1}K=a_{s+1}L=0$.
\end{lemma}
\begin{proof}
It is easy to see that $K$ is a submodule of $M$. Also, finitely generated modules over a Noetherian ring are always Noetherian, so $M$ is a Noetherian $A$-module and $K$ is finitely generated. For each $n\in\mathbb{N}$, $K_n$ is the kernel of the map $m\mapsto{a_{s+1}}m$, so $a_{s+1}{K}=a_{s+1}\oplus_{n=0}^{\infty}K_n=0$. $L$ is finitely generated because it is a quotient of $M$, and it is trivial that $a_{s+1}L=0$.
\end{proof}

\begin{lemma}
\label{KLfg}
$K$ and $L$ are both finitely generated $A_0[a_1,\ldots,a_{s}]$-modules.
\end{lemma}
\begin{proof}
Because $K$ is a finitely generated module over $A=A_0[a_1,\ldots,a_{s+1}]$, there exist $k_1,\ldots,k_l\in{K}$ that generate $K$. Given any $k\in{K}$, $k=c_1{k_1}+\cdots+c_l{k_l}$ for some $c_1,\ldots,c_l\in{A}$. Each $c_i$ can be written as $c_i(a_1,\ldots,a_{s+1})$, where $c_i$ is a polynomial over the ring $A_0$. If we write $c_i{k_i}=b_1{k_i}+\cdots+b_{r_i}{k_i}$, where $r_i$ is a number dependent on $i$ and each $b_j$ is a monomial, then for any $b_j$ that contains $a_{s+1}$ as a factor, $b_j{k_i}=0$, because we have proved that $a_{s+1}{K}$ is the zero module. These arguments have shown that if we write the expression $c_1{k_1}+\cdots+c_l{k_l}$ explicitly (i.e. expand all brackets and combine all like terms), then any term that contains the factor $a_{s+1}$ will vanish. As $k$ was chosen arbitrarily, this implies that $K$ is actually a finitely generated ${A_0}[a_1,\ldots,a_{s}]$-module. Similarly, $L$ is a finitely generated module over ${A_0}[a_1,\ldots,a_{s}]$ as well.
\end{proof}

\begin{proof}[inductive step]
Note that for any $n\in\mathbb{N}$, we get an exact sequence
\begin{equation}
\label{ExactSeq}
0\rightarrow{K_n}\rightarrow{M_n}\stackrel{g_n}{\rightarrow}{M_{n+d_{s+1}}}\rightarrow{L_{n+d_{s+1}}}\rightarrow{0}.
\end{equation}
As $A_0$ is Noetherian, the kernel of any homomorphism between two finitely generated $A_0$-modules is finitely generated. Hence, Lemma \ref{sumeq0ofexact} applies to (\ref{ExactSeq}) and we have
\begin{equation}
\mu(K_n)-\mu(M_n)+\mu(M_{n+d_{s+1}})-\mu(L_{n+d_{s+1}})=0.
\end{equation}
Multiplying by $t^{n+d_{s+1}}$ and summing over $n$, we obtain
\begin{equation}
\begin{split}
& t^{d_{s+1}}\sum_{n=0}^{\infty}t^{n}\mu(K_n)-t^{d_{s+1}}\sum_{n=0}^{\infty}t^{n}\mu(M_n)+
\\
& \sum_{n=0}^{\infty}t^{n+d_{s+1}}\mu(M_{n+d_{s+1}})-\sum_{n=0}^{\infty}t^{n+d_{s+1}}\mu(L_{n+d_{s+1}})=0.
\end{split}
\end{equation}
It is easy to see that the above implies that
\begin{equation}
(1-t^{d_{s+1}})P(M,\mu,t)+t^{d_{s+1}}P(K,\mu,t)-P(L,\mu,t)=p(t)
\end{equation}
for some polynomial $p(t)$. Multiplying by $\Pi_{i=1}^{s}(1-t^{d_i})$, we know that
\begin{equation}
\begin{split}
& P(M,\mu,t)\prod_{i=1}^{s+1}(1-t^{d_i})+t^{d_{s+1}}P(K,\mu,t)\prod_{i=1}^{s}(1-t^{d_i})
-
\\
& P(L,\mu,t)\prod_{i=1}^{s}(1-t^{d_i})=q(t)
\end{split}
\end{equation}
for some polynomial $q(t)$. $K$ and $L$ are finitely generated $A_0[a_1,\ldots,a_{s}]$-modules; moreover, since $A_0$ is Noetherian by Lemma \ref{gradezeroNoering}, $A_0[a_1,\ldots,a_{s}]$ is a Noetherian graded ring. Hence, we can use the induction hypothesis to deduce that the expressions $t^{d_{s+1}}P(K,\mu,t)\Pi_{i=1}^{s}(1-t^{d_i})$ and $P(L,\mu,t)\Pi_{i=1}^{s}(1-t^{d_i})$ are both polynomials, meaning that $P(M,\mu,t)\Pi_{i=1}^{s+1}(1-t^{d_i})$ is also a polynomial, so we are done.
\end{proof}
